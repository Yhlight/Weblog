GDScript-for循环与in
2025-6-11

for-in

for-in.gd 

                    
extends Node

func _ready() -> void:
	# for循环与常数
	for i in 3:
		print(i)
		
	print()
		
	# for循环与range
	for i in range(10, 0, -1):
		print(i)
		
	print()
	
	# for循环与字符串
	var str: String = "HelloWorld"
	for i in str:
		print(i)
		
	print()
		
	# for循环与数组
	var arr: Array[String] = ["荧火light", "没睡醒的扬", "幻来听"]
	for i in arr:
		print(i)
		
	print()
		
	# for循环与字典
	var dic: Dictionary[String, int] = {
		"荧火light": 18,
		"没睡醒的扬": 19,
		"幻来听": 18
	}
	for key in dic:
		print(dic[key])
