命名类

命名类.gd

                    
# 使用class_name命名本文件
class_name YhLight

var yhlight: Dictionary[String, int] = {
	"荧火light": 18,
	"幻来听": 18
}

func show() -> void:
	var arr: Array[String]
	
	for key in yhlight:
		arr.push_back(key)
	
	print(arr)

命名类测试.gd

                    
extends Node

# 命名类也不允许使用const创建对象
var yhLight: YhLight = YhLight.new()

func _ready() -> void:
	print(yhLight.yhlight)
	
	yhLight.show()
