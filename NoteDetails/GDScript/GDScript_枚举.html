2025-6-11

                    
extends Node

# 枚举常用于快速创建一连串的常量
# 枚举拥有与字典相同的方法
enum yhlight {
	YHLIGHT,
	SHUIYANG,
	HUANLAITING
}

enum state {
	STATE_IDLE = 5,
	STATE_JUMP = 10,
	STATE_SHOOT = 15
}

func _ready() -> void:
	# 使用枚举名.KEY来访问枚举常量
	print(yhlight.YHLIGHT)
	print(state.STATE_IDLE)
	
	# for循环与枚举
	for key in yhlight:
		printt(key, yhlight[key])
		
	print()
	
	for key in state:
		printt(key, state[key])
